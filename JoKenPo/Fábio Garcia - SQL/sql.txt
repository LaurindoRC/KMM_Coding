/*
1 - Pedra
2 - Papel
3 - Tesoura
*/
select decode(t.user_value, 1, 'Pedra', 2, 'Papel', 3, 'Tesoura', 'Burro, não tem essa opção') as "Você escolheu"
     , decode(t.ia_value, 1, 'Pedra', 2, 'Papel', 3, 'Tesoura', 'Burro, não tem essa opção') as "PC escolheu"
     , case
          when t.user_value = t.ia_value then 'Ninguem ganha porra nenhuma'
          when t.user_value = 1 and t.ia_value = 2 then 'PC ganhou, vacilão'
          when t.user_value = 1 and t.ia_value = 3 then 'Usuário ganhou, PC burro'
          when t.user_value = 2 and t.ia_value = 1 then 'Usuário ganhou, PC burro'
          when t.user_value = 2 and t.ia_value = 3 then 'PC ganhou, vacilão'
          when t.user_value = 3 and t.ia_value = 1 then 'PC ganhou, vacilão'
          when t.user_value = 3 and t.ia_value = 2 then 'Usuário ganhou, PC burro'
       end as mensagem
  from (select &user_value as user_value
             , round(dbms_random.value(1,3)) as ia_value
          from dual
       ) t
